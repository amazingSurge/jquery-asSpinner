$(".custom").asSpinner();

mousewheel / keyboard


$(".mousewheel").asSpinner({
    mousewheel: true
});

custom format


$(".format").asSpinner({
    format: function(value) {
        return '$'+value;
    }
});

custom rule


$(".rule").asSpinner({
    rule: 'month' // day month hour minute second
});
