demo

Default


$(".custom").asSpinner();

mousewheel / keyboard


$(".mousewheel").asSpinner({
    mousewheel: true
});

custom format

Note: format option should be set with parse options, or plugin will run error


$(".format").asSpinner({
    format: function(value) {
        return '$'+value;
    },
    parse: function(value) {
        return value.replace('$','');
    }
});

custom rule


$(".rule").asSpinner({
    rule: 'month' // day month hour minute second
});
